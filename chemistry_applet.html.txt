let factorCount = 0;

function addFactor() {
    factorCount++;

    const container = document.getElementById("factorContainer");

    const block = document.createElement("div");
    block.id = "factor" + factorCount;
    block.style.marginBottom = "12px";
    block.style.padding = "10px";
    block.style.border = "1px solid #1e3a5f";
    block.style.background = "#11151c";

    block.innerHTML = `
        <strong>Factor ${factorCount}</strong><br>
        <input id="num${factorCount}" type="number" step="any" placeholder="numerator" style="width:100px;">
        <input id="numUnit${factorCount}" type="text" placeholder="unit" style="width:100px;">
        &nbsp; / &nbsp;
        <input id="den${factorCount}" type="number" step="any" placeholder="denominator" style="width:100px;">
        <input id="denUnit${factorCount}" type="text" placeholder="unit" style="width:100px;">
        &nbsp;
        <button onclick="removeFactor(${factorCount})">Remove</button>
    `;

    container.appendChild(block);
}

function removeFactor(id) {
    const block = document.getElementById("factor" + id);
    if (block) block.remove();
}

function calculateDA() {
    let value = parseFloat(document.getElementById("daValue").value);
    let unit = document.getElementById("daUnit").value.trim();

    if (isNaN(value) || unit === "") {
        document.getElementById("daOutput").innerText = "Enter a starting value and unit.";
        return;
    }

    let output = `Starting Value:\n${value} ${unit}\n\nSteps:\n`;

    let currentValue = value;
    let currentUnit = unit;

    for (let i = 1; i <= factorCount; i++) {
        const block = document.getElementById("factor" + i);
        if (!block) continue;

        let num = parseFloat(document.getElementById("num" + i).value);
        let numUnit = document.getElementById("numUnit" + i).value.trim();
        let den = parseFloat(document.getElementById("den" + i).value);
        let denUnit = document.getElementById("denUnit" + i).value.trim();

        if (isNaN(num) || isNaN(den) || numUnit === "" || denUnit === "") {
            output += `Factor ${i}: incomplete — skipped.\n`;
            continue;
        }

        // Multiply value
        currentValue = currentValue * (num / den);

        // Unit cancellation
        if (currentUnit === denUnit) {
            currentUnit = numUnit;
        } else {
            currentUnit = `${currentUnit}·(${numUnit}/${denUnit})`;
        }

        output += `× (${num} ${numUnit} / ${den} ${denUnit}) → ${currentValue} ${currentUnit}\n`;
    }

    output += `\nFinal Answer:\n${currentValue} ${currentUnit}`;

    document.getElementById("daOutput").innerText = output;
}

function loadGasInputs() {
    let mode = document.getElementById("gasMode").value;
    let box = document.getElementById("gasInputs");

    if (mode === "pvnrt") {
        box.innerHTML = `
            P (atm): <input id="P" type="number" step="any"><br>
            V (L): <input id="V" type="number" step="any"><br>
            n (mol): <input id="n" type="number" step="any"><br>
            T (K): <input id="T" type="number" step="any"><br>
            <small>Leave ONE blank to solve for it.</small>
        `;
    }

    if (mode === "combined") {
        box.innerHTML = `
            P1: <input id="P1" type="number" step="any">
            V1: <input id="V1" type="number" step="any">
            T1: <input id="T1" type="number" step="any"><br>
            P2: <input id="P2" type="number" step="any">
            V2: <input id="V2" type="number" step="any">
            T2: <input id="T2" type="number" step="any"><br>
        `;
    }

    if (mode === "dalton") {
        box.innerHTML = `
            P_total: <input id="Ptot" type="number" step="any"><br>
            P1: <input id="P1d" type="number" step="any"><br>
            P2: <input id="P2d" type="number" step="any"><br>
        `;
    }

    if (mode === "density") {
        box.innerHTML = `
            Density (g/L): <input id="D" type="number" step="any"><br>
            Pressure (atm): <input id="Pden" type="number" step="any"><br>
            Molar Mass (g/mol): <input id="M" type="number" step="any"><br>
            Temperature (K): <input id="Tden" type="number" step="any"><br>
        `;
    }
}

function calculateGas() {
    let mode = document.getElementById("gasMode").value;
    let out = "";

    if (mode === "pvnrt") {
        const R = 0.08206;
        let P = parseFloat(P.value);
        let V = parseFloat(V.value);
        let nVal = parseFloat(n.value);
        let Tval = parseFloat(T.value);

        if (isNaN(P)) P = (nVal * R * Tval) / V;
        else if (isNaN(V)) V = (nVal * R * Tval) / P;
        else if (isNaN(nVal)) nVal = (P * V) / (R * Tval);
        else if (isNaN(Tval)) Tval = (P * V) / (nVal * R);

        out = `PV = nRT\nP=${P}, V=${V}, n=${nVal}, T=${Tval}`;
    }

    if (mode === "combined") {
        let P1v = parseFloat(P1.value);
        let V1v = parseFloat(V1.value);
        let T1v = parseFloat(T1.value);
        let P2v = parseFloat(P2.value);
        let V2v = parseFloat(V2.value);
        let T2v = parseFloat(T2.value);

        out = `Combined Gas Law\n(P1V1/T1) = (P2V2/T2)\nInputs: ${P1v}, ${V1v}, ${T1v}, ${P2v}, ${V2v}, ${T2v}`;
    }

    if (mode === "dalton") {
        let Ptot = parseFloat(Ptot.value);
        let P1d = parseFloat(P1d.value);
        let P2d = parseFloat(P2d.value);

        out = `Dalton's Law\nP_total = P1 + P2\nInputs: ${Ptot}, ${P1d}, ${P2d}`;
    }

    if (mode === "density") {
        let Dv = parseFloat(D.value);
        let Pv = parseFloat(Pden.value);
        let Mv = parseFloat(M.value);
        let Tv = parseFloat(Tden.value);

        out = `Density Gas Law\nD = PM/RT\nInputs: ${Dv}, ${Pv}, ${Mv}, ${Tv}`;
    }

    document.getElementById("gasOutput").innerText = out;
}

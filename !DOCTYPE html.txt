let factorCountDA = 0;

function addFactorDA() {
    factorCountDA++;
    const div = document.createElement("div");
    div.className = "factor";
    div.innerHTML = `
        <strong>Factor ${factorCountDA}</strong>
        <input id="num${factorCountDA}" type="number" step="any" placeholder="Numerator">
        <input id="numUnit${factorCountDA}" type="text" placeholder="Unit">
        /
        <input id="den${factorCountDA}" type="number" step="any" placeholder="Denominator">
        <input id="denUnit${factorCountDA}" type="text" placeholder="Unit">
    `;
    document.getElementById("factorsDA").appendChild(div);
}

function calculateDA() {
    let value = parseFloat(document.getElementById("startValueDA").value);
    let unit = document.getElementById("startUnitDA").value.trim();
    let steps = `Starting value:\n${value} ${unit}\n\nSteps:\n`;

    for (let i = 1; i <= factorCountDA; i++) {
        let num = parseFloat(document.getElementById("num" + i).value);
        let numUnit = document.getElementById("numUnit" + i).value.trim();
        let den = parseFloat(document.getElementById("den" + i).value);
        let denUnit = document.getElementById("denUnit" + i).value.trim();

        if (!num || !den || !numUnit || !denUnit) continue;

        value *= (num / den);

        let shownUnit = (unit === denUnit) ? numUnit : unit + "·" + numUnit + "/" + denUnit;

        steps += `${unit} × (${num} ${numUnit} / ${den} ${denUnit}) → ${value.toFixed(6)} ${shownUnit}\n`;

        unit = shownUnit;
    }

    document.getElementById("outputDA").innerText = steps;
}

const periodic = {
    H:1.008, He:4.003, Li:6.94, Be:9.012, B:10.81, C:12.01, N:14.01, O:16.00, F:19.00, Ne:20.18,
    Na:22.99, Mg:24.31, Al:26.98, Si:28.09, P:30.97, S:32.06, Cl:35.45, Ar:39.95,
    K:39.10, Ca:40.08, Fe:55.85, Cu:63.55, Zn:65.38, Br:79.90, Ag:107.87, I:126.90, Ba:137.33, Hg:200.59, Pb:207.2
};

function parseFormula(formula) {
    let stack = [{}];
    let i = 0;

    while (i < formula.length) {
        if (formula[i] === "(") {
            stack.push({});
            i++;
        } else if (formula[i] === ")") {
            let group = stack.pop();
            i++;
            let num = "";
            while (/[0-9]/.test(formula[i])) num += formula[i++];
            num = num ? parseInt(num) : 1;

            for (let el in group) {
                stack[stack.length - 1][el] = (stack[stack.length - 1][el] || 0) + group[el] * num;
            }
        } else {
            let el = formula[i++];
            if (/[a-z]/.test(formula[i])) el += formula[i++];

            let num = "";
            while (/[0-9]/.test(formula[i])) num += formula[i++];
            num = num ? parseInt(num) : 1;

            stack[stack.length - 1][el] = (stack[stack.length - 1][el] || 0) + num;
        }
    }
    return stack[0];
}

function molarMass(formula) {
    let parsed = parseFormula(formula);
    let mass = 0;
    for (let el in parsed) {
        mass += periodic[el] * parsed[el];
    }
    return mass;
}

function calculateStoich() {
    let eq = document.getElementById("equation").value;
    let givenVal = parseFloat(document.getElementById("givenValue").value);
    let givenUnit = document.getElementById("givenUnit").value;
    let givenSub = document.getElementById("givenSubstance").value.trim();
    let targetSub = document.getElementById("targetSubstance").value.trim();
    let targetUnit = document.getElementById("targetUnit").value;

    let [left, right] = eq.split("->").map(s => s.trim());
    let reactants = left.split("+").map(s => s.trim());
    let products = right.split("+").map(s => s.trim());

    function getCoeff(species) {
        let match = species.match(/^(\d+)\s*(.*)$/);
        return match ? [parseInt(match[1]), match[2]] : [1, species];
    }

    let coeffs = {};
    [...reactants, ...products].forEach(s => {
        let [c, f] = getCoeff(s);
        coeffs[f] = c;
    });

    let steps = `Balanced equation: ${eq}\n\n`;

    let mmGiven = molarMass(givenSub);
    let mmTarget = molarMass(targetSub);

    steps += `Molar mass of ${givenSub}: ${mmGiven.toFixed(3)} g/mol\n`;
    steps += `Molar mass of ${targetSub}: ${mmTarget.toFixed(3)} g/mol\n\n`;

    let molesGiven = givenUnit === "g" ? givenVal / mmGiven :
                     givenUnit === "mol" ? givenVal :
                     givenUnit === "L" ? givenVal / 22.4 :
                     givenUnit === "particles" ? givenVal / 6.022e23 : 0;

    steps += `Convert given to moles: ${molesGiven.toFixed(6)} mol ${givenSub}\n`;

    let moleRatio = coeffs[targetSub] / coeffs[givenSub];
    let molesTarget = molesGiven * moleRatio;

    steps += `Apply mole ratio: × (${coeffs[targetSub]} / ${coeffs[givenSub]}) = ${molesTarget.toFixed(6)} mol ${targetSub}\n`;

    let final;
    if (targetUnit === "g") final = molesTarget * mmTarget;
    if (targetUnit === "mol") final = molesTarget;
    if (targetUnit === "L") final = molesTarget * 22.4;
    if (targetUnit === "particles") final = molesTarget * 6.022e23;

    steps += `\nFinal answer: ${final.toExponential(6)} ${targetUnit}`;

    document.getElementById("outputStoich").innerText = steps;
}
